#include <iostream>
using namespace std;
int main() {
    float celsiu, fahrenheit;

//Solicitar al usuario el valor de la temperatura en grados Celsius    
    cout << "Ingrese la temperatura en grados Celsius: ";
    cin >> celsiu;

//Realizar la conversión de grados Celsius a grados Fahrenheit    
    fahrenheit = (celsiu * 9/5) + 32;

//Imprimir el resultado de la conversión    
    cout << "La temperatura en grados Fahrenheit es = " << fahrenheit << endl;
    
    return 0;
}
